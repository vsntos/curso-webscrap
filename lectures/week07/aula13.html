NLP: Natural Language Processing

Conjunto de técnicas para extrair estrutura e significado de textos.

Por que quanteda?

O pacote de referência para NLP em R

  • desenvolvido por Kenneth Benoit (LSE);
  • muito usado em Ciência Política;
  • eficiente com grandes corpora;
  • integra com tidyverse.
install.packages("quanteda")
install.packages("quanteda.textstats")
install.packages("quanteda.textplots")

O Pipeline NLP

Fluxo padrão

texto bruto
    ↓
corpus (quanteda)
    ↓
tokens
    ↓
DFM (Document-Feature Matrix)
    ↓
análise

Corpus

Objeto central do quanteda

library(quanteda)
library(tidyverse)

titulos <- c(
  "ONU debate reforma do Conselho de Segurança",
  "BRICS avança em moeda alternativa ao dólar",
  "OTAN amplia presença militar no Leste Europeu",
  "Brasil e China assinam acordos comerciais",
  "Irã rejeita proposta nuclear do Ocidente"
)

corp <- corpus(
  titulos,
  docnames = paste0("doc", 1:5)
)

corp

Adicionar metadados

datas <- as.Date(c(
  "2026-05-12", "2026-05-11", "2026-05-10",
  "2026-05-09", "2026-05-08"
))

docvars(corp, "data")  <- datas

summary(corp)

Tokenização

Quebrar texto em tokens

toks <- tokens(
  corp,
  remove_punct    = TRUE,
  remove_numbers  = TRUE,
  remove_symbols  = TRUE
)

toks[1]

Remover stopwords

stopwords_pt <- c(
  stopwords("pt"),
  "que", "com", "para", "por", "mais", "uma", "dos",
  "das", "nos", "nas", "seu", "sua", "são", "está"
)

toks_limpos <- tokens_remove(toks, stopwords_pt)

toks_limpos[1]

Ngrams: combinações de palavras

toks_ngrams <- tokens_ngrams(toks_limpos, n = 2)

toks_ngrams[1:2]

Keyness: Comparação de Corpora

Quais palavras distinguem grupos?

titulos_seg <- c(
  "ONU debate reforma multilateral",
  "Brasil lidera negociações climáticas",
  "BRICS amplia acordos econômicos",
  "OTAN anuncia novas bases militares",
  "EUA sancionam aliados do Kremlin",
  "Conflito no Oriente Médio se intensifica"
)

grupo <- c("sul", "sul", "sul", "norte", "norte", "norte")

corp_seg <- corpus(titulos_seg)
docvars(corp_seg, "grupo") <- grupo

dfm_seg <- corp_seg |>
  tokens(remove_punct = TRUE) |>
  tokens_remove(stopwords("pt")) |>
  dfm() |>
  dfm_group(groups = grupo)

keyness <- textstat_keyness(dfm_seg, target = "sul")
head(keyness)

Visualizar keyness

textplot_keyness(keyness)

Dicionários

Classificar textos por tema

dicionario_ri <- dictionary(list(
  multilateralismo = c("onu", "otan", "brics", "g20",
                        "multilateral", "conselho", "assembléia"),
  conflito         = c("guerra", "conflito", "sanções",
                        "militar", "ataque", "defesa"),
  economia         = c("comércio", "exportação", "pib",
                        "tarifa", "moeda", "dólar")
))

dfm_dict <- dfm_lookup(dfm_corp, dictionary = dicionario_ri)

convert(dfm_dict, to = "data.frame")

Sentiment: Polaridade

Classificar tom positivo / negativo

# Dicionário de valência para português
dict_valence <- dictionary(list(
  positivo = c("acordo", "cooperação", "parceria",
               "paz", "progresso", "avanço"),
  negativo = c("conflito", "sanção", "crise",
               "tensão", "ruptura", "guerra")
))

dfm_sent <- dfm_lookup(dfm_corp, dictionary = dict_valence)

# Calcular escore
scores <- convert(dfm_sent, to = "data.frame") |>
  mutate(escore = positivo - negativo)

scores
