O que Pode Dar Errado

Tudo

  • servidor fora do ar;
  • timeout de conexão;
  • mudança de estrutura HTML;
  • rate limiting (429);
  • redirecionamentos inesperados;
  • elementos ausentes na página;
  • HTML malformado;
  • encoding inconsistente.

Um script frágil quebra na primeira falha

Um script robusto:

  • registra o erro;
  • continua para o próximo item;
  • tenta de novo quando faz sentido;
  • entrega o que conseguiu coletar.

tryCatch()

A ferramenta central

resultado <- tryCatch(
  {
    # Código que pode falhar
    log(-1)
  },
  warning = function(w) {
    cat("Aviso:", conditionMessage(w), "\n")
    NA
  },
  error = function(e) {
    cat("Erro:", conditionMessage(e), "\n")
    NULL
  }
)

resultado

Aplicado ao scraping

pagina <- tryCatch(
  read_html(url),
  error = function(e) {
    message("Falha ao acessar: ", url)
    return(NULL)
  }
)

if (is.null(pagina)) return(tibble())

possibly() e safely()

purrr oferece wrappers convenientes

library(purrr)

# possibly: retorna valor padrão em caso de erro
ler_pagina_segura <- possibly(
  read_html,
  otherwise = NULL
)

# safely: retorna lista com $result e $error
ler_pagina_detalhada <- safely(read_html)

Usando possibly()

urls <- c(
  "https://agenciabrasil.ebc.com.br/internacional",
  "https://url-que-nao-existe.xyz/pagina"
)

paginas <- map(urls, possibly(read_html, otherwise = NULL))

# Filtrar sucessos
paginas_validas <- compact(paginas)

Usando safely()

resultados <- map(urls, safely(read_html))

# Separar resultados e erros
paginas <- map(resultados, "result")
erros   <- map(resultados, "error")

# Quais falharam?
failed <- map_lgl(erros, ~!is.null(.x))
urls[failed]

Retentativas

Tentar de novo em caso de falha temporária

raspar_com_retry <- function(url, max_tentativas = 3) {

  for (tentativa in seq_len(max_tentativas)) {

    pagina <- tryCatch(
      read_html(url),
      error = function(e) NULL
    )

    if (!is.null(pagina)) return(pagina)

    # Espera exponencial entre tentativas
    delay <- 2 ^ tentativa
    cat(sprintf(
      "Tentativa %d falhou. Aguardando %ds...\n",
      tentativa, delay
    ))
    Sys.sleep(delay)
  }

  message("Todas as tentativas falharam: ", url)
  return(NULL)
}

httr2 tem retry embutido

library(httr2)

resp <- request(url) |>
  req_retry(
    max_tries = 5,
    backoff   = ~ 2 ^ .x  # 2, 4, 8, 16s
  ) |>
  req_perform()

Logging de Erros

Registrar o que deu errado

log_file <- "logs/scraping_erros.txt"
dir.create("logs", showWarnings = FALSE)

log_erro <- function(url, mensagem) {
  linha <- sprintf(
    "[%s] ERRO em %s: %s\n",
    format(Sys.time(), "%Y-%m-%d %H:%M:%S"),
    url,
    mensagem
  )
  cat(linha)
  cat(linha, file = log_file, append = TRUE)
}

# Usar no loop
pagina <- tryCatch(
  read_html(url),
  error = function(e) {
    log_erro(url, e$message)
    return(NULL)
  }
)

Validar Resultados

Verificar antes de usar

validar_pagina <- function(pagina, seletor = "h2") {

  if (is.null(pagina)) return(FALSE)

  elementos <- html_elements(pagina, seletor)

  if (length(elementos) == 0) {
    message("Nenhum elemento encontrado para: ", seletor)
    return(FALSE)
  }

  TRUE
}

Verificar dataset resultante

validar_dataset <- function(df, n_min = 5) {

  if (nrow(df) < n_min) {
    warning(sprintf(
      "Dataset muito pequeno: %d observações (mínimo: %d)",
      nrow(df), n_min
    ))
    return(FALSE)
  }

  if (any(is.na(df$titulo))) {
    warning("Títulos com NA encontrados.")
  }

  TRUE
}

Scraping Ético: o Que Não Fazer

Comportamentos a evitar

  • requisições sem delay;
  • parallelismo agressivo;
  • ignorar 429;
  • contornar CAPTCHAs;
  • falsificar identidade.

Comportamentos corretos

request(url) |>
  # Identificar corretamente
  req_user_agent("Pesquisa acadêmica RI / vinicius@uni.br") |>

  # Respeitar rate limits
  req_throttle(rate = 10 / 60) |>

  # Retry apenas em erros transitórios
  req_retry(max_tries = 3) |>

  req_perform()

Script Robusto Completo

Estrutura recomendada

raspar_pagina_robusta <- function(url) {

  # 1. Delay respeitoso
  Sys.sleep(runif(1, 1, 2))

  # 2. Requisição com retry e timeout
  resp <- tryCatch(
    request(url) |>
      req_user_agent("Curso RI") |>
      req_timeout(15) |>
      req_retry(max_tries = 3) |>
      req_perform(),
    error = function(e) {
      log_erro(url, e$message)
      return(NULL)
    }
  )

  if (is.null(resp)) return(tibble())
  if (resp_status(resp) != 200) return(tibble())

  # 3. Parse
  pagina <- resp |>
    resp_body_string() |>
    read_html()

  # 4. Extração segura
  titulos <- tryCatch(
    pagina |> html_elements("h2") |> html_text2(),
    error = function(e) character(0)
  )

  if (length(titulos) == 0) return(tibble())

  # 5. Retornar
  tibble(titulo = titulos, url_fonte = url, coletado = Sys.time())
}
