# Exemplo de Pipeline

library(tidyverse)
library(xml2)
library(rvest)
library(lubridate)


# 1. Função: busca RSS do Google News ------------------------------------
get_google_news <- function(query) {
  
  url <- paste0(
    "https://news.google.com/rss/search?q=",
    URLencode(query),
    "&hl=pt-BR&gl=BR&ceid=BR:pt-419"
  )
  
  rss <- tryCatch(read_xml(url), error = function(e) NULL)
  if (is.null(rss)) return(tibble())
  
  items <- xml_find_all(rss, "//item")
  if (length(items) == 0) return(tibble())
  
  lc_orig <- Sys.getlocale("LC_TIME")
  Sys.setlocale("LC_TIME", "C")
  on.exit(Sys.setlocale("LC_TIME", lc_orig))
  
  tibble(
    title    = xml_text(xml_find_first(items, "title")),
    link     = xml_text(xml_find_first(items, "link")),
    pub_date = xml_text(xml_find_first(items, "pubDate")),
    source   = xml_text(xml_find_first(items, "source")),
    query    = query
  ) %>%
    mutate(
      pub_date = as.POSIXct(
        strptime(
          sub(" [A-Z]+$", "", pub_date),  # remove "GMT" antes de parsear
          "%a, %d %b %Y %H:%M:%S",
          tz = "UTC"
        )
      )
    )
}

# 2. Queries -------------------------------------------------------------

queries <- c(
  "diplomacia",
  "Irã",
  "Trump"
)


# 3. Coleta e filtro por data --------------------------------------------
# Sem after:/before: na URL — o RSS retorna os ~100 resultados mais recentes
# e filtramos por pub_date aqui

df_news <- purrr::map_dfr(queries, get_google_news) %>%
  filter(
    !is.na(pub_date),
    pub_date >= as.POSIXct("2025-04-10", tz = "UTC"),
    pub_date <  as.POSIXct("2026-04-16", tz = "UTC")
  ) %>%
  distinct(link, .keep_all = TRUE)   # remove duplicatas entre queries
